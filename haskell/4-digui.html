    indirect enum List<T: CustomStringConvertible>: CustomStringConvertible {
    case null
    case cons(T, List<T>)
    
    init(list: [T]) {
        if list.count == 0 {
            self = .null
        } else {
            let trail = Array(list.suffix(from: 1))
            self = .cons(list[0], .init(list: trail))
        }
    }
    
    var description: String {
        switch self {
        case .null:
            return "null"
        case let .cons(a, b):
            return "cons(\(a), \(b.description))"
        }
    }
    
    func toList() -> [T] {
        
        switch self {
        case .null: return []
        case let .cons(a, b):
            return [a] + b.toList()
        }
    }
}
